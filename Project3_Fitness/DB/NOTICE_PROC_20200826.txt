PROCEDURE PROC_NOTICE_FILTER_LIST
  (
    O_CUR OUT SYS_REFCURSOR
  ) AS
  BEGIN
    OPEN O_CUR FOR
    SELECT MENU_ID, MENU_NAME, MENU_SEQ
    FROM MENUS;
    
  END PROC_NOTICE_FILTER_LIST;

PROCEDURE PROC_NOTICE_INSERT
  (
    IN_MENUID IN CHAR,
    IN_TITLE   IN VARCHAR2,
    IN_CONT    IN VARCHAR2,
    IN_MID   IN VARCHAR2,
    IN_BNUM   IN NUMBER,
    IN_LVL   IN NUMBER,
    IN_STEP IN NUMBER,
    IN_NREF IN NUMBER,
    --파일 업로드에 대한 추가 항목 
    IN_FILENAMES IN FILE_ARRAY,
    IN_FILEEXTS  IN FILE_ARRAY,
    IN_SFILENAMES IN FILE_ARRAY
  ) AS
  
    V_BNUM      NUMBER(5, 0);
    V_LVL       NUMBER(5, 0);
    V_STEP      NUMBER(5, 0);
    V_NREF      NUMBER(5, 0);
    V_MAXIDX    NUMBER(5, 0);
  
  BEGIN
    
    IF IN_BNUM = 0 THEN --새글을 위한 변수처리 
    SELECT NVL(MAX(BNUM),0) + 1
    INTO V_BNUM FROM BOARD
    WHERE MENU_ID = IN_MENUID;
    
    V_LVL := 0;
    V_STEP := 0;
    
    SELECT NVL( MAX(NREF), 0 ) + 1 
    INTO V_NREF FROM BOARD
    WHERE MENU_ID = IN_MENUID;
    
    ELSE    -- 답글 처리
            V_BNUM  := IN_BNUM;
            V_LVL   := IN_LVL + 1;
            V_STEP  := IN_STEP + 1;
            V_NREF  := IN_NREF;
            
            -- 기존 답글에 대한 순서 + 1
            UPDATE  BOARD
            SET     STEP    = STEP + 1
            WHERE   MENU_ID  = IN_MENUID
            AND     NREF    = IN_NREF
            AND     STEP    > IN_STEP
            ;
        END IF;
    
    INSERT INTO BOARD (IDX, MENU_ID, TITLE, CONT, MID,
    REGDATE, READCOUNT, BNUM, LVL, STEP, NREF, LIKECOUNT)
    VALUES (BRDSEQ.NEXTVAL, IN_MENUID, IN_TITLE, IN_CONT, IN_MID,
    TO_CHAR(SYSDATE, 'YYYY-MM-DD HH:MI:SS'), 0, 
    V_BNUM, V_LVL, V_STEP, V_NREF, 0);
    
    -- 가장 마지막에 추가 된 글의 IDX 구하는 쿼리
    SELECT MAX(IDX) INTO V_MAXIDX FROM BOARD;
    
    -- 여러 FILE 처리 
    IF IN_FILENAMES(1) IS NOT NULL THEN -- 배열의 첫번째 -> 오라클에서는 1번 부터 
    -- 첫번째에 파일이 있는지 확인 
    FOR I IN IN_FILENAMES.FIRST .. IN_FILENAMES.LAST
    LOOP 
     INSERT INTO FILES
     VALUES (
     (SELECT NVL(MAX(FILE_NUM), 0) +1 FROM FILES),
     V_MAXIDX, -- 마지막에 추가된 게시글 번호 
     IN_FILENAMES(I),
     IN_FILEEXTS(I),
     IN_SFILENAMES(I));
    END LOOP;
  
    END IF;
    
  END PROC_NOTICE_INSERT;

  PROCEDURE PROC_NOTICE_PLUS_READCOUNT
 (
    IN_IDX  IN  NUMBER  
 ) AS
  BEGIN
    UPDATE BOARD
    SET READCOUNT = READCOUNT +1
    WHERE IDX = IN_IDX;
 
  END PROC_NOTICE_PLUS_READCOUNT;

  PROCEDURE PROC_NOTICE_GETCONTENT
 (
    IN_IDX IN NUMBER,
    O_CUR OUT SYS_REFCURSOR
 ) AS
  BEGIN
   
   OPEN O_CUR FOR
   SELECT T1.MNAME, T2.REGDATE, T3.MENU_NAME, T2.TITLE, T2.CONT, TRIM(T2.MENU_ID) AS MENU_ID
    FROM MEMBERS T1, BOARD T2, MENUS T3
    WHERE T1.MID = T2.MID
    AND     T2.MENU_ID = T3.MENU_ID
    AND     T2.IDX = IN_IDX;
   
   
  END PROC_NOTICE_GETCONTENT;

  PROCEDURE PROC_NOTICE_HOWLIKE
 (
    IN_MID IN VARCHAR2,
    IN_IDX  IN NUMBER,
    O_CUR OUT SYS_REFCURSOR
 ) AS
 

 
  BEGIN
 
 OPEN O_CUR FOR
   SELECT ISLIKE
    FROM BLIKE
    WHERE MID = IN_MID
    AND IDX = IN_IDX;
  END PROC_NOTICE_HOWLIKE;

  PROCEDURE PROC_NOTICE_FILELIST
 (
    IN_IDX IN NUMBER,
    O_CUR OUT SYS_REFCURSOR
 ) AS
  BEGIN
   OPEN O_CUR FOR
   SELECT T1.FILE_NUM, T1.IDX, T1.FILENAME, T1.FILEEXT, T1.SFILENAME,T3.MID
   FROM FILES T1, BOARD T2, MEMBERS T3
   WHERE T1.IDX = T2.IDX 
   AND T2.MID = T3.MID
   AND T1.IDX = IN_IDX;
   
  END PROC_NOTICE_FILELIST;

  PROCEDURE PROC_NOTICE_UPDATE
 (
    IN_TITLE IN VARCHAR2,
    IN_CONT IN VARCHAR2,
    IN_IDX IN NUMBER,
    IN_MENUID IN CHAR,
    --파일 업로드에 대한 추가 항목 
    IN_FILENAMES IN FILE_ARRAY,
    IN_FILEEXTS  IN FILE_ARRAY,
    IN_SFILENAMES IN FILE_ARRAY
 ) AS
  BEGIN
    
    UPDATE BOARD
    SET TITLE = IN_TITLE, CONT = IN_CONT, MENU_ID = IN_MENUID
    WHERE IDX = IN_IDX;
    
      -- 여러 FILE 처리 
    IF IN_FILENAMES(1) IS NOT NULL THEN -- 배열의 첫번째 -> 오라클에서는 1번 부터 
    -- 첫번째에 파일이 있는지 확인 
    FOR I IN IN_FILENAMES.FIRST .. IN_FILENAMES.LAST
    LOOP 
     INSERT INTO FILES
     VALUES (
     (SELECT NVL(MAX(FILE_NUM), 0) +1 FROM FILES),
     IN_IDX,
     IN_FILENAMES(I),
     IN_FILEEXTS(I),
     IN_SFILENAMES(I));
    END LOOP;
  
    END IF;
    
  END PROC_NOTICE_UPDATE;

  PROCEDURE PROC_NOTICE_DELETE
 (
    IN_IDX IN NUMBER
 ) AS
  BEGIN
    DELETE 
    FROM BOARD
    WHERE IDX = IN_IDX;
    
    DELETE
    FROM FILES
    WHERE IDX = IN_IDX;
    
  END PROC_NOTICE_DELETE;
